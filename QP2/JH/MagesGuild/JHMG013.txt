Clock _timer_ 00:00 0 flag 17 range 0 2
Clock _find_ 00:06 00:22

--	Quest start-up:
	start timer _timer_
    pc at _city_ set _arrival_
    log 1030 step 0
    get item _elem_

_arrival_ task:
    say 1010
    start timer _find_

_stop_ task:
    when not _arrival_
    stop timer _find_

_find_ task:
    say 1012
    create foe _atr_ every 0 minutes 1 times with 100% success

_fail_ task:
    killed _atr_
    say 1020
    end quest

_stun_ task:
    injured _atr_
    say 1013
    restrain foe _atr_

_qgclicked_ task:
	clicked npc _qgiver_ 

_clearclick_ task:
	when _qgclicked_ and not _stun_ 
	clear _qgclicked_ _clearclick_ 

_pcreward_ task:
	when _qgclicked_ and _stun_ 
    take _elem_ from pc
    give pc _reward_
	end quest 

_timer_ task:
    say 1003
	end quest
