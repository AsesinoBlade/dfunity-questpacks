QBN:
Clock _timer_ 1.00:00
Clock _failsafe_ 20.00:00

Foe _enemy_ is Vampire
Item _reward_ magic_item

--	Quest start-up:
    start timer _failsafe_

_stealth_ task:
    when skill Stealth is at least 70

_thau_ task:
    when skill Thaumaturgy is at least 50

_level_ task:
    level 5 completed

_time_ task:
    daily from 3:00 to 3:30

_enter_ task:
    when pc enters village
    clear _exit_

_exit_ task:
    when pc exits village
    clear _enter_

_meet_ task:
    when _time_ and _enter_ and _level_
    say 1010
    create foe _enemy_ every 0 minutes 1 times with 100% success
    give item _reward_ to _enemy_
    make _reward_ permanent
    start timer _timer_
    start task _spawn_

variable _spawn_
_safe1_ task:
    when _spawn_ and _thau_
    say 1011
    restrain foe _enemy_

_safe2_ task:
    when _spawn_ and _stealth_
    say 1012
    restrain foe _enemy_

_slain_ task:
    killed _enemy_
    end quest

_timer_ task:
    end quest

_failsafe_ task:
    end quest
