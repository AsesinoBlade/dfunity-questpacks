QBN:
Clock _timer_ 0:05
Clock _failsafe_ 90.00:00

Foe _enemy_ is Zombie

Item _gem_ gem

--	Quest start-up:
    start timer _failsafe_

_begin_ task:
    when pc enters graveyard
    prompt 1010 yes _try_ no _quit_
    have _gem_ set _attempt_
    start timer _timer_

_quit_ task:
    say 1012

_try_ task:
    give pc _gem_

_attempt_ task:
    pick one of _free_ _trap_

variable _free_
_trap_ task:
    say 1011
    create foe _enemy_ every 0 minutes 2 times with 100% success

_timer_ task:
    end quest

_failsafe_ task:
    end quest
