Person _lhotun_ named Prince_Lhotun atHome

Clock _letterclock_ 45.00:00 50.00:00
Clock _destroyletter_ 3.00:00
Clock _timer_ 0:01
Clock _timer2_ 0:02

Item _letter_ letter used 1011

--	Quest start-up:
	start timer _letterclock_

_letterclock_ task:
    give pc _letter_ notify 1010
    have _letter_ set _startdestroy_

_startdestroy_ task:
    start timer _destroyletter_

_destroyletter_ task:
    take _letter_ from pc

_clicklhotun_ task:
    clicked npc _lhotun_

_notyet_ task:
    when _clicklhotun_ and not _letterclock_
    clear _clicklhotun_ _notyet_

_talklhotun_ task:
    when _clicklhotun_ and _letterclock_
    prompt 1012 yes _begin_ no _refuse_

_refuse_ task:
    say 1026
    end quest

_begin_ task:
    say 1013
    hide npc _lhotun_
    add _lhotun_ face
    pc at any armory set _insidefg_
    pc at _palace_ set _insidepalace_
    change repute with _lhotun_ by +10

variable _insidepalace_

_talk1_ task:
    when _begin_ and not _insidepalace_ and not _lhotunleaves_
    say 1014
    log 1015 step 0

_insidefg_ task:
    start task _lhotunleaves_

-- this was going to be inside a fighters guild, but
--(1) sentinel city doesn't have any, and
--(2) in hammerfell there are no fighters guilds, only
-- fighters trainers, which I can't detect,
-- so the idea was moot to begin with.
-- big sad, sentinel city is a very bad city

_lhotunleaves_ task:
    say 1016
    drop _lhotun_ face
    start timer _timer_
    start task _soundeffects_
    change repute with _lhotun_ by +5

_soundeffects_ task:
    play sound sword_defend_2 every 0 minutes 1 times
    play sound sword_defend_1 every 0 minutes 1 times

_timer_ task:
    say 1018
    clear _soundeffects_
    add _lhotun_ face
    start timer _timer2_
    remove log step 0

variable _timer2_

_talk2_ task:
    when _timer2_ and not _insidefg_
    say 1019
    log 1020 step 0
    pc at any library set _insidelibr_

_insidelibr_ task:
    start task _talk3_

_talk3_ task:
    say 1021
    log 1022 step 0
    change repute with _lhotun_ by +5

_return_ task:
    when _insidepalace_ and _talk3_
    prompt 1023 yes _continue_ no _endquest_
    restore npc _lhotun_

_endquest_ task:
    say 1025
    end quest

_continue_ task:
    give pc nothing
    start quest TTClhotun2
    end quest
