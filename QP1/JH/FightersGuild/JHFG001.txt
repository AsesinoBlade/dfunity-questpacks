QBN:
Item _reward_ gold

Person _qgiver_ group Questor male

Place _vampres_ remote house1

Clock _timer_ 00:00 0 flag 17 range 0 2

Foe _vamp_ is Vampire
Foe _simp_ is Knight

--	Quest start-up:
	start timer _timer_
	log 1030 step 0
	pick one of _real_ _false_ _real_

_real_ task:
	place foe _vamp_ at _vampres_

_slain_ task:
	killed 1 _vamp_ saying 1020
	start task _victory_

variable _victory_
_pcgetsgold_ task:
	when _qgclicked_ and _slain_
	give pc _reward_
	end quest

_false_ task:
	place foe _simp_ at _vampres_
	restrain foe _simp_

_inj_ task:
	injured _simp_
	prompt 1010 yes _escape_ no _ignore_

_escape_ task:
	say 1011
	remove foe _simp_

variable _ignore_
_murder_ task:
	killed _simp_
	say 1012
	start task _victory_

_qgclicked_ task:
	clicked npc _qgiver_

_clearclick_ task:
	when _qgclicked_ and not _victory_ and not _escape_
	clear _qgclicked_ _clearclick_

_convince_ task:
	when _qgclicked_ and _escape_
	prompt 1013 yes _begin_ no _fail2_

_begin_ task:
	pick one of _go1_ _go2_ _go3_

_check1_ task:
	when skill Streetwise is at least 15

_check2_ task:
	when skill Streetwise is at least 25

_check3_ task:
	when skill Streetwise is at least 35

variable _go1_
variable _go2_
variable _go3_

_c1_ task:
	when _check1_ and _go1_
	start task _pcgetsgold2_

_c2_ task:
	when _check2_ and _go2_
	start task _pcgetsgold2_

_c3_ task:
	when _check3_ and _go3_
	start task _pcgetsgold2_

_d1_ task:
	when not _check1_ and _go1_
	start task _fail_

_d2_ task:
	when not _check2_ and _go2_
	start task _fail_

_d3_ task:
	when not _check3_ and _go3_
	start task _fail_

_pcgetsgold2_ task:
	give pc _reward_
	end quest

_fail_ task:
	say 1014
	start task _fail2_

_fail2_ task:
	say 1015
	end quest

_timer_ task:
	say 1003
	end quest
