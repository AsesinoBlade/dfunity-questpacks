--	Quest start-up:
	start timer _timer_ 
	log 1030 step 0 
    reveal _dung_
    add _friend_ face
    pc at _dung_ do _inside_
    pick one of _e1_ _e2_

_inside_ task:
    say 1019
    create foe _enemy1_ every 101 minutes indefinitely with 10% success
    create foe _enemy2_ every 32 minutes indefinitely with 2% success

_slain_ task:
	killed 3 _enemy1_
	killed 3 _enemy2_

variable _e1_
_true_ task:
    when _slain_ and _e1_
    say 1020
    drop _friend_ face
    give pc _reward_
    end quest

variable _e2_
_trick_ task:
    when _slain_ and _e2_
    prompt 1021 yes _mercy_ no _kill_

_mercy_ task:
    say 1022
    give pc nothing
    end quest

_kill_ task:
    drop _friend_ face
    say 1023
    create foe _frenemy_ every 0 minutes 1 times with 100% success
    give item _reward_ to _frenemy_

_kill2_ task:
    killed _frenemy_
    make _reward_ permanent

_qgclicked_ task:
	clicked npc _qgiver_ 

_clearclick_ task:
	when _qgclicked_ and not _kill2_ 
	clear _qgclicked_ _clearclick_ 

_question_ task:
    when _qgclicked_ and _kill2_
    say 1025
    change repute with _fg_ by +3
    end quest
